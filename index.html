string[,] logs = new string [50, 5];

byte choice = 0;

bool error = false;

do

{

Console.Write("Главное меню:\n" +

" 1 Добавить задачу\n" +

" 2 Удалить задачу\n" +

" 3 Редактировать задачу\n" +

" 4 Просмотреть список задач\n" +

" 5 Выйти из приложения\n" +

"Ввод: ");

choice = byte.Parse(Console.ReadLine());

switch (choice)

{

case 1:

Console.WriteLine("Добавить задачу:");

for (int i = 0; i !!!= logs.GetLength(0); i++)

{

if (i == 50)

{

error = true;

break;

}

if (logs[i, 0] == null)

{

logs[i, 0] = $"{i + 1}";

Console.Write("Введите название: ");

logs[i, 1] = Console.ReadLine();

Console.Write("В формате дд.мм.гггг или ISO yyyy-mm-dd\n" +

$"Сейчас: {DateTime.Now.ToString("d")}\n" +

"Введите дату: ");

logs[i, 2] = Console.ReadLine();

Console.Write("Формат: чч:мм\n" +

$"Сейчас: {DateTime.Now.ToString("t")}\n" +

"Введите время: ");

logs[i, 3] = Console.ReadLine();

Console.Write("Приоритет (целое число, например, 1 — высокий, 2 — средний, 3 — низкий)\n" +

"Введите: ");

logs[i, 4] = Console.ReadLine();

break;

}

else

continue;

}

break;

}

if (error)

{

Console.WriteLine($"[{DateTime.Now.ToString("g")}] Превышено максимальное количество логов. Новые записи не сохраняются");

}

}

while (true);
